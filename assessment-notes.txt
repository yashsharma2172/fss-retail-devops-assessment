-- ArgoCD Install Steps --
  kubectl create namespace argocd
  kubectl apply -n argocd -f https://raw.githubusercontent.com/argoproj/argo-cd/stable/manifests/install.yaml
  kubectl create namespace murali-ns
  kubectl patch svc argocd-server -n argocd -p '{"spec": {"type": "LoadBalancer"}}'
  kubectl port-forward svc/argocd-server -n argocd 8080:443
  argocd admin initial-password -n argocd
  argocd login <ARGOCD-LB-URL>:80
  argocd account update-password

-- ArgoCD Application points to --
  repoURL: https://github.com/Msocial123/fss-Retail-App_kubernetes.git
  path: k8s-manifests
  namespace: murali-ns
  syncPolicy: automated (prune: true, selfHeal: true)

TASK 4: Datadog Monitoring for FSS Retail App
==============================================================
-- Docx Requirements --
  [✓] Cluster health, node utilization, pod status, container metrics, K8s events
  [✓] App availability, response latency, transaction failures, errors, resource consumption
  [✓] Application + Container + Kubernetes + Error logs
  [✓] Dashboards: Retail Operations + Infrastructure
  [✓] Alerts: downtime, pod restart, CPU, memory, errors, performance degradation

-- Install Datadog Agent → see datadog-setup.sh --
  helm repo add datadog https://helm.datadoghq.com
  helm install datadog-agent datadog/datadog \
    --set datadog.apiKey=$DATADOG_API_KEY \
    --set datadog.logs.enabled=true \
    --set datadog.logs.containerCollectAll=true \
    --set datadog.apm.enabled=true \
    --set datadog.processAgent.enabled=true

-- Retail Operations Dashboard widgets --
  1. App availability   : service check on retail-app-service endpoint
  2. Transaction perf   : HTTP request rate + p95 response time
  3. Error trends       : 5xx error rate over time

-- Infrastructure Dashboard widgets --
  1. EKS cluster health : node status, ready vs not-ready
  2. CPU utilization    : per pod and per node
  3. Memory utilization : per pod, alert if > 85%
  4. Pod availability   : pod count vs desired replicas

-- Alerts to configure --
  - retail-app-service down        → CRITICAL (synthetics check)
  - pod restarts > 3 in 10min      → CRITICAL (kubernetes.containers.restarts)
  - CPU > 80%                      → WARNING  (system.cpu.user)
  - Memory < 15% free              → WARNING  (system.mem.pct_usable)
  - HTTP error rate > 2%           → CRITICAL (trace.servlet.request.errors)
  - Response time > 2s             → WARNING  (trace.servlet.request.duration)

==============================================================
-- Docx Requirements --

  [✓] Claude AI integration for log analysis, error ID, RCA, recommendations, incident summary
  [✓] Scenario 1: App unavailable after deployment
  [✓] Scenario 2: Slow checkout response
  [✓] Scenario 3: Pods continuously restarting
  [✓] Scenario 4: Production errors increase after new release

-- See ai-agent.py for full code --

-- Scenario 1: Retail App unavailable after deployment --
  kubectl get pods -n murali-ns                                  ← deployment status
  kubectl describe deployment retail-app-deployment -n murali-ns ← pod health
  kubectl logs <pod-name> -n murali-ns                           ← application logs
  kubectl get events -n murali-ns --sort-by=.lastTimestamp       ← K8s events
  Claude → ImagePullBackOff / wrong port / ConfigMap missing / Secret not found

-- Scenario 2: Slow checkout response --
  kubectl top pods -n murali-ns                                  ← resource utilization
  kubectl top nodes                                              ← error patterns
  kubectl logs <pod-name> -n murali-ns | grep -i "slow\|error"  ← performance bottlenecks
  Claude → DB connection timeout / CPU throttling / missing index in MongoDB

-- Scenario 3: Pods continuously restarting (CrashLoopBackOff) --
  kubectl get pods -n murali-ns
  kubectl describe pod <pod-name> -n murali-ns                   ← CrashLoopBackOff events
  kubectl logs <pod-name> -n murali-ns --previous                ← container logs
  Claude → OOMKilled / wrong env var / MongoDB not ready / bad image tag

-- Scenario 4: Production errors increase after new release --
  kubectl rollout history deployment/retail-app-deployment -n murali-ns  ← deployment timeline
  kubectl logs <pod-name> -n murali-ns | grep -i error                   ← error logs
  kubectl get events -n murali-ns                                         ← recent changes
  Claude → code regression / new config error / dependency mismatch
  Fix → kubectl rollout undo deployment/retail-app-deployment -n murali-ns
